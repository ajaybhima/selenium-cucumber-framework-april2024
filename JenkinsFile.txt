pipeline {
    agent any
 
    stages {
        stage('Test') {
            steps {
                bat mvn -D clean test -Dcucumber.filter.tags="@SmokeTest"
            }
        }    
        stage('Publish Cucumber Reports') {
        steps {            
                   publishHTML([
                       allowMissing: false, 
                       alwaysLinkToLastBuild: false, 
                       keepAll: false, 
                       reportDir: 'target/', 
                       reportFiles: 'cucumber-reports.html', 
                       reportName: 'HTML Report', 
                       reportTitles: '', 
                       useWrapperFileDirectly: true])
                }
            }
    }
    post {
            always {
                // Optional: Clean up workspace if needed
                cleanWs()
            }
    }   
}
